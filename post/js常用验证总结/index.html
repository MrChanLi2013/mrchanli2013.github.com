js常用验证总结

2017-09-11
JavaScript
约 350 字 预计阅读 1 分钟

文章目录

一些常用的js正则验证：

example：

var validateRegExp = {
    chinese: "^[\\u4e00-\\u9fa5]+$" //仅中文
};

var validateRules = {
 isChinese: function (str) {
    return new RegExp(validateRegExp.chinese).test(str);
  }
};

validateRules.isChinese(str);

总结：

  • decmal: “^([+-]?)\d*\.\d+$“, //浮点数
  • decmal1: “1\d.\d|0.\d[1-9]\d$“, //正浮点数
  • decmal2: “^-([1-9]\d.\d|0.\d[1-9]\d)$“, //负浮点数
  • intege: “^-?[1-9]\d*$“, //整数
  • intege1: “2\d*$“, //正整数
  • intege2: “^-[1-9]\d*$“, //负整数
  • num: “^(?:-?\d+|-?\d{1,3}(?:,\d{3})+)?(?:.\d+)?$“, //数字
  • num1: “3\d*|0$“, //正数（正整数 + 0）
  • num2: “^-[1-9]\d*|0$“, //负数（负整数 + 0）
  • ascii: “4+$“, //仅ACSII字符
  • chinese: “5+$“, //仅中文
  • color: “6{6}$“, //颜色
  • date: “^\d{4}(\-|\/|.)\d{1,2}\1\d{1,2}$“, //日期
  • email: “7+@[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?(?:\.[a-zA-Z0-9](?:[a-zA-Z0-9-]{0,61}[a-zA-Z0-9])?)*$“, //邮箱
  • idcard: “^[1-9]([0-9]{14}|[0-9]{17})$“, //身份证
  • ip4: “^(25[0-5]|2[0-4]\d|[0-1]\d{2}|[1-9]?\d)\.(25[0-5]|2[0-4]\d|[0-1]\d{2}|[1-9]?\d)\.(25[0-5* 0-4]\d|[0-1]\d{2}|[1-9]?\d)\.(25[0-5]|2[0-4]\d|[0-1]\d{2}|[1-9]?\d)$“, //ip地址
  • letter: “8+$“, //字母
  • letter_l: “9+$“, //小写字母
  • letter_u: “10+$“, //大写字母
  • mobile: “^0?(13|15|18|14)[0-9]{9}$“, //手机
  • notempty: “^\S+$“, //非空
  • password: “^.*[A-Za-z0-9\w_-]+.*$“, //密码
  • password_: “^[^ ]{6,16}$“, //不能含有空格的密码
  • fullNumber: “11+$“, //数字
  • picture: “(.*)\.(jpg|bmp|gif|ico|pcx|jpeg|tif|png|raw|tga)$“, //图片
  • qq: “^\[1-9]*[1-9][0-9]*$“, //QQ号码
  • rar: “(.*)\.(rar|zip|7zip|tgz)$“, //压缩文件
  • tel: “12{2,20}$“, //电话号码的函数(包括验证国内区号,国际区号,分机号)
  • url: “^(?:(?:(?:https?|ftp):)?\/\/)(?:\S+(?::\S*)?@)?(?:(?!(?:10|127)(?:\.\d{1,3}){3})(?!(?:169\.254|192\.168)(?:\.\d{1,3}){2})(?!172\.(?:1[6-9]|2\d|3[0-1])(?:\.\d{1,3}){2})(?:[1-9]\d?|1\d\d|2[01]\d|22[0-3])(?:\.(?:1?\d{1,2}|2[0-4]\d|25[0-5])){2}(?:\.(?:[1-9]\d?|1\d\d|2[0-4]\d|25[0-4]))|(?:(?:[a-z\u00a1-\uffff0-9]-*)*[a-z\u00a1-\uffff0-9]+)(?:.(?:[a-z\u00a1-\uffff0-9]-*)*[a-z\u00a1-\uffff0-9]+)*(?:.(?:[a-z\u00a1-\uffff]{2,})).?)(?::\d{2,5})?(?:[\/?#]\S*)?$“, //url
  • username: “13{2,25}$“, //户名
  • zipcode: “^\d{6}$“, //邮编
  • realname: “14+$“, // 真实姓名
  • ……
